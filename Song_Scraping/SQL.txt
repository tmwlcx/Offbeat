-- Setup. DO NOT REMOVE.
.headers on
.separator ','

CREATE TABLE artist (
    id TEXT PRIMARY KEY,
    name TEXT,
    uri TEXT,
    followers INTEGER
);

CREATE TABLE album (
    album_id TEXT PRIMARY KEY,
    artist_id TEXT,
    album_name TEXT,
    album_uri TEXT
);

CREATE TABLE song (
    song_id TEXT PRIMARY KEY,
    artist_id TEXT,
    album_id TEXT,
    danceability REAL,
    energy REAL,
    key INT,
    loudness REAL,
    mode INT,
    speechiness REAL,
    acousticness REAL,
    instrumentalness REAL,
    liveness REAL,
    valence REAL,
    tempo REAL
);
